\begin{displaymath}
O\mleft(\group{\frac{n}{g}}^k (kg)^3 \log^3 (kg)\mright)
\end{displaymath}

\begin{displaymath}
kg
\end{displaymath}}

O\Bigg(n (kn^{1-\frac{1}{k}})^3 \log^3 (kn^{1-\frac{1}{k}})\Bigg)
\end{displaymath}
\bblue{Query size}
\begin{displaymath}
kn^{1-\frac{1}{k}}
