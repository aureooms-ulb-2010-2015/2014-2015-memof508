\section{\defineconcept{Comparability Graphs}}
\subsection{Definition}

The comparability graph $\text{G}(P)$ of a POSet $P$ is an undirected graph where all comparable elements of the POSet are adjacent.

\subsection{Cliques and Stable Sets}

\defbox{A set S is a clique if the vertices in S are pairwise adjacent.}
\defbox{A set S is stable if the vertices in S are pairwise nonadjacent.}


\subsection{\defineconcept{STAB($G$)}}
\begin{equation}
\text{STAB}(G) := \text{conv}\{\chi^S \in R^V : S\text{ stable set in }G\}
\label{eq:stab}
\end{equation}

\subsection{Chains and Anti-Chains}

\subsection{\defineconcept{Entropy for Comparability Graphs}}
\begin{equation}
\text{H}(G) := \min_{x \in \text{STAB}(G)}~ -\frac{1}{n} \sum_{v \in V} \log x_v
\label{eq:entropy:graph}
\end{equation}

\subsection{\defineconcept{Entropy for POSets}}
\begin{equation}
\text{H}(P) := \text{H}(\text{G}(P))
\label{eq:entropy:poset}
\end{equation}
